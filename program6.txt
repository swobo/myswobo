
#include <stdio.h>

{
        int a[20],i,n;
        int info,pom;
        do
        {
                printf("\n Podaj iloqwqwsc elementow tablicy a: \n n = ");
                scanf("%d",&n);
        }
        while(n<1||n>20);

        {
                printf(" Podaj a[%d] = ",i);
                scanf("%d",&a[i]);
        }

        {
