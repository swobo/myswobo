
#include "stdio.h"
#include "conio.h"
main()
{
        float temp[100],min,maks;
        int i,ilosc;
        printf("\n Podaj ilosc dni: \n n = ");
        scanf("%d",&ilosc);
        printf("\n Podaj %d temperatur: \n\n",ilosc);
        for(i=0;i<ilosc;i++)
        {
                printf(" %d TEMPERATURA: \t\t ",i+1);
                scanf("%f",&temp[i]);
        }
        min=temp[0];
        maks=temp[0];
        for(i=1;i<ilosc;i++)
        {
                if(temp[i] < min)
                        min=temp[i];
                if(temp[i] > maks)
                        maks=temp[i];
        }
        printf("\n MINIMALNA TEMPERATURA: \t %f",min);
        printf("\n MAKSYMALNA TEMPERATURA: \t %f",maks);
        getch();
        return 0;
}
