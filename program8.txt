
#include <stdio.h>

int main()
{
  int array[100], maximum, wielkosc, c, location = 1;

  printf("Wpisz liczbe elementow w tablicy\n");
  scanf("%d", &wielkosc);

  printf("Wprowadz %d liczby calkowite\n", wielkosc);

  for (c = 0; c < wielkosc; c++)
    scanf("%d", &array[c]);

  maximum = array[0];

  for (c = 1; c < wielkosc; c++)
  {
    if (array[c] > maximum)
    {
       maximum  = array[c];
       location = c+1;
    }
  }

  printf("Element o najwiekqwqwszej wartosci zostal wprowadzony na pozycji %d i jego wartosc wynosi %d.\n", location, maximum);
  return 0;
}
